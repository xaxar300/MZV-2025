integer function abs(integer x)
{
    declare integer r;
    if (x < 0) {
        r = 0 - x;
    } else {
        r = x;
    };
    return r;
};

integer function max(integer a, integer b)
{
    declare integer r;
    if (a > b) {
        r = a;
    } else {
        r = b;
    };                   
    return r;                      
};

main
{
    declare integer a;
    declare integer b;
    declare integer c;
    declare char ch;

    a = 25;
    b = 10;
    ch = 'X';

    output 'Привет';
    output a;
    output b;

    c = a + b;
    output c;

    if (a > b) {
        output 'больше';
    } else {
        output 'меньше';
    };

    c = max(a, b);
    output c;

    output 'Конец';
    return 0;
};
